<pre><code>public static boolean isPalindrome(final char[] palindromeString,final int len){

    if( null  == palindromeString || len &lt;1){
        return false;
    }

    int middle = middlePos(len);

    int before = middle -1;

    int behind = (len % 2 == 0) ? middle : (middle + 1);

    while(before &gt;= 0){
        System.out.println(palindromeString[before]+"----"+palindromeString[behind]);
        if(palindromeString[before--] != palindromeString[behind++]){
            return false;
        }
    }
    return true;


}

private static int middlePos(final int num ){
    
    int middle = (num &gt;&gt; 1);

    return middle &gt; 0 ? middle : 0;
}
