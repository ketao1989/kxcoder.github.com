<p>def lowerLetter(c):
    if c &gt;= ‘a’ and c &lt;= ‘z’:
        return True
    else:
        return False</p>

<p>def firstUpper(s):
    ss = ‘’
    for i in xrange(0, len(s)):
        if i == 0:
            if lowerLetter(s[i]):
                ss = ss + chr(ord(s[i]) + ord(‘A’) - ord(‘a’))
            else:
                ss = ss + s[i]</p>
