public void run() {
        Throwable exception = null;
        try {
            onStart(); // 该方法主要负责初始化Task 队列，和seed值
            pool.work(this); // 调用pool的work方法，在pool中说明
        } catch (Throwable ex) {
            exception = ex;
        } finally {
            onTermination(exception);// 清除该worker线程关于结束的一些操作，比如取消任务，解除在pool上的注册，状态为结束terminate
        }
    }
