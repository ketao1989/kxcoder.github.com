def closure_tranform(ll):
    res = []
    for x in ll:
        def tranform():
            xf = x + '-tranform'
            return xf
        res.append(tranform)
    return res

# Azc-tranform Azc-tranform Azc-tranform
lf1, lf2, lf3 = closure_tranform(['abc', 'zfc', 'Azc'])
print lf1, lf2, lf3
print lf1(), lf2(), lf3()
