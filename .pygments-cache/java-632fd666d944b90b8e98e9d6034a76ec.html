void initTable(AtomicReferenceArray&lt;ReferenceEntry&lt;K, V&gt;&gt; newTable) {
    this.threshold = newTable.length() * 3 / 4; // 0.75
    if (!map.customWeigher() &amp;&amp; this.threshold == maxSegmentWeight) {
        // prevent spurious expansion before eviction
        this.threshold++;
    }
    this.table = newTable;
}
