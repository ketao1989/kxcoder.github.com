def defaultParams(x, n=2, extra=0):
    multiply = 1
    while n > 0:
        n -= 1
        multiply = multiply * x
    return multiply + extra


def variableParams(*numbers):
    sum1 = 0
    for num in numbers:
        sum1 += num
    return sum1
